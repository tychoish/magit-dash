* Overview

=magit-dash= is a tabulated-list dashboard for tracking multiple local git
repositories from a single buffer.  Each registered repo shows its current
branch, fetch age, ahead/behind counts, dirty state, and configured
auto-sync operations.  Selecting a repo opens a per-repo overview buffer with
recent log, PR counts (via the GitHub CLI), and magit action shortcuts.

A companion PR dashboard (=magit-dash-gh-pr=) and CI log fetcher
(=magit-dash-gh-actions=) are provided as optional modules.

Repositories are registered with =magit-dash-register= (typically in your
=user/= or personal init file) and the dashboard is opened with
=magit-dash-open=.

* Installation
** package-vc.el

Requires Emacs 29.1+.  Run once to clone and register the package:

#+BEGIN_SRC emacs-lisp
(package-vc-install '(magit-dash
                      :url "https://github.com/tychoish/magit-dash"))
#+END_SRC

** use-package with VC

Requires Emacs 30+.  Installs on first load if not already present:

#+BEGIN_SRC emacs-lisp
(use-package magit-dash
  :vc (:url "https://github.com/tychoish/magit-dash"))
#+END_SRC

** use-package with load-path

#+BEGIN_SRC sh
git clone https://github.com/tychoish/magit-dash ~/path/to/magit-dash
#+END_SRC

#+BEGIN_SRC emacs-lisp
(use-package magit-dash
  :load-path "~/path/to/magit-dash"
  :commands (magit-dash-open magit-dash-open-other-window))
#+END_SRC

  (save-excursion
    (org-back-to-heading t)
    (let ((end (save-excursion (org-end-of-subtree t) (point))))
      (forward-line 1)
      (while (and (< (point) end) (not (org-at-heading-p)))
        (forward-line 1))
      (when (< (point) end)
        (delete-region (point) end))))

  (let* ((org-dir (file-name-directory (buffer-file-name)))
         ;; All .el files in the package, excluding test files.
         (el-files (seq-sort
                    #'string<
                    (seq-remove
                     (lambda (f) (string-match-p "-test\\.el$\\|test-.*\\.el$\\|/test/" f))
                     (directory-files org-dir t "\\.el$" t))))
         ;; Scope: controls which definitions are documented.
         ;;   'exported   -- all non-private public symbols (no -- in name)
         ;;   'autoloaded -- only ;;;###autoload-annotated symbols
         ;;   'interactive -- all interactive commands (commandp)
         ;;   (list ...)  -- explicit named symbols, flat output in given order
         (scope 'autoloaded)
         ;; Additional symbol kinds to include beyond what scope selects.

    (defun magit-dash-doc--section-name (comment-text)
      (let* ((name (string-trim (car (split-string comment-text " -- "))))
             (name (replace-regexp-in-string "-" " " name)))
        (mapconcat (lambda (s) (if (string-match-p "\\." s) s (capitalize s)))
                   (split-string name "[ \t]+" t) " ")))

    (defun magit-dash-doc--format-doc (doc)
      (let* ((paras (split-string doc "\n\n" t))
             (fn-para (seq-find (lambda (p) (string-match "\\`(fn " (string-trim p))) paras))
             (rest-paras (seq-remove (lambda (p) (string-match "\\`(fn " (string-trim p))) paras)))
        (mapconcat #'identity
                   (delq nil
                         (list
                          (when fn-para (concat ": " (string-trim fn-para)))
                          (when rest-paras
                            (mapconcat
                             (lambda (para)
                               (if (string-match "\\`[ \t]" para)
                                   para
                                 (mapconcat #'string-trim (split-string para "\n" t) " ")))
                             rest-paras "\n\n"))))
                   "\n\n")))

    (defun magit-dash-doc--defkind (def-keyword)
      (cond
       ((member def-keyword '("defvar" "defconst" "defvar-local")) 'variable)
       ((equal def-keyword "defcustom") 'custom)
       (t 'function)))

    (defun magit-dash-doc--kind-tag (sym kind)
      (cond
       ((commandp sym) " [Command]")
       ((eq kind 'custom) " [Option]")
       ((eq kind 'variable) " [Variable]")
       (t "")))

    (defun magit-dash-doc--format-sym (name kind &optional heading)
      (let* ((sym (intern name))
             (fn-p (fboundp sym))
             (var-p (and (not fn-p) (boundp sym)))
             (doc (or (when fn-p (documentation sym))
                      (when var-p (documentation-property sym 'variable-documentation)))))
        (format "%s~%s~%s\n\n%s\n\n"
                (concat (or heading "*** ") (if doc "" "TODO "))
                name
                (magit-dash-doc--kind-tag sym kind)
                (if doc (magit-dash-doc--format-doc doc) "*no docstring*"))))

    (defun magit-dash-doc--include-p (name kind autoload-p)
      (and (not (string-match-p "[a-z]--" name))
           (cond
            ((eq kind 'variable) (memq 'variables include-kinds))
            ((eq kind 'custom)
             (or (memq 'customs include-kinds)
                 (and (memq scope '(exported autoloaded)) autoload-p)))
            (t
             (or (eq scope 'exported)
                 (and (eq scope 'autoloaded) autoload-p)
                 (eq scope 'interactive))))))

    (cond
     ;; Section-based collection for 'exported, 'autoloaded, and 'interactive.
     ;; Sections delimited by long ;;;;... ruler lines; the first ;; comment
     ;; after each ruler names the section.
     ((memq scope '(exported autoloaded interactive))
      (let (sections current-section current-syms after-ruler autoload-next)
        (seq-do
         (lambda (el-file)
           (with-temp-buffer
             (insert-file-contents el-file)
             (goto-char (point-min))
             (while (not (eobp))
               (let ((line (buffer-substring-no-properties
                            (line-beginning-position) (line-end-position))))
                 (cond
                  ((string-match "^;\\{10,\\}" line)
                   (when (and current-section current-syms)
                     (push (cons current-section (nreverse current-syms)) sections))
                   (setq current-section nil current-syms nil after-ruler t))
                  ((and after-ruler (string-match "^;; \\(.+\\)" line))
                   (setq current-section (magit-dash-doc--section-name (match-string 1 line))
                         after-ruler nil))
                  ((string-match "^;;;###autoload" line)
                   (setq autoload-next t))
                  ((string-match "^(\\(?:cl-\\)?\\(def[^ ]+\\) +'?\\([^ ()]+\\)" line)
                   (let* ((def-kw (match-string 1 line))
                          (sym-name (match-string 2 line))
                          (kind (magit-dash-doc--defkind def-kw)))
                     (when (and current-section
                                (magit-dash-doc--include-p sym-name kind autoload-next))
                       (push (list sym-name kind) current-syms))
                     (setq autoload-next nil)))
                  ((not (string-empty-p (string-trim line)))
                   (setq autoload-next nil))))
               (forward-line 1)))
           (when (and current-section current-syms)
             (push (cons current-section (nreverse current-syms)) sections))
           (setq current-section nil current-syms nil after-ruler nil autoload-next nil))
         el-files)
        (seq-do
         (lambda (section)
           (let ((syms (if (eq scope 'interactive)
                           (seq-filter (lambda (e) (commandp (intern (car e)))) (cdr section))
                         (cdr section))))
             (when syms
               (princ (format "** %s\n\n" (car section)))
               (seq-do (lambda (entry)
                         (princ (magit-dash-doc--format-sym (car entry) (cadr entry))))
                       syms))))
         (nreverse sections))))

     ;; Flat output for an explicit symbol list, in the given order.
     ((listp scope)
      (seq-do (lambda (name)
                (princ (magit-dash-doc--format-sym name 'function "** ")))
              scope))))
